import numpy as np
import matplotlib.pyplot as plt


def load_ekg_signal(file_name="ekg100.txt"):
    """
    Wczytuje sygna EKG z pliku.
    Dla ekg100.txt zakada si czstotliwo prbkowania 360 Hz.
    """
    try:

        data = np.loadtxt(f'sygnaly-testowe/{file_name}')
    except FileNotFoundError:
        print(f"Bd: Plik '{file_name}' nie znaleziono w folderze 'sygnaly-testowe/'.")
        print("Upewnij si, e plik jest w odpowiedniej lokalizacji.")
        return None, None, None

    signal = data

    sampling_rate = 360
    time = np.linspace(0, len(signal) / sampling_rate, len(signal), endpoint=False)

    print(f"Pomylnie wczytano sygna '{file_name}'.")

    return time, signal, sampling_rate


def compute_and_plot_fft(signal, sampling_rate, title="Widmo amplitudowe sygnau EKG"):
    """
    Wyznacza dyskretn transformat Fouriera sygnau i przedstawia
    widmo amplitudowe w zakresie czstotliwoci [0, fs/2].
    """
    N = len(signal)
    freqs = np.fft.fftfreq(N, 1 / sampling_rate)
    fft_values = np.fft.fft(signal)
    fft_magnitude = np.abs(fft_values) / N

    positive_freq_indices = np.where(freqs >= 0)
    freqs_positive = freqs[positive_freq_indices]
    fft_magnitude_positive = fft_magnitude[positive_freq_indices]

    plt.figure(figsize=(12, 6))
    plt.plot(freqs_positive, fft_magnitude_positive)
    plt.title(title)
    plt.xlabel("Czstotliwo [Hz]")
    plt.ylabel("Amplituda")
    plt.grid(True)
    plt.xlim(0, sampling_rate / 2)
    plt.tight_layout()
    plt.show()

def compute_ifft_and_compare(original_time, original_signal, fft_values, plot_samples_limit=1000):
    """
    Wyznacza odwrotn dyskretn transformat Fouriera i porwnuje
    z pierwotnym sygnaem. Wywietla fragmenty sygnaw.
    """
    reconstructed_signal = np.fft.ifft(fft_values).real


    plot_len = min(len(original_signal), plot_samples_limit)

    plt.figure(figsize=(12, 6))
    plt.plot(original_time[:plot_len], reconstructed_signal[:plot_len])
    plt.title("Zrekonstruowany sygna (Odwrotna transformata Fouriera - Fragment)")
    plt.xlabel("Czas [s]")
    plt.ylabel("Amplituda")
    plt.grid(True)
    plt.tight_layout()
    plt.show()

    plt.figure(figsize=(12, 6))
    plt.plot(original_time[:plot_len], original_signal[:plot_len], label="Oryginalny sygna", alpha=0.9)
    plt.plot(original_time[:plot_len], reconstructed_signal[:plot_len], label="Zrekonstruowany sygna", linestyle="dashed", alpha=0.7)
    plt.title("Porwnanie oryginalnego i zrekonstruowanego sygnau EKG (Fragment)")
    plt.xlabel("Czas [s]")
    plt.ylabel("Amplituda")
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.show()


    difference_signal = original_signal - reconstructed_signal
    plt.figure(figsize=(12, 6))
    plt.plot(original_time[:plot_len], difference_signal[:plot_len])
    plt.title("Rnica midzy oryginalnym a zrekonstruowanym sygnaem EKG (Fragment)")
    plt.xlabel("Czas [s]")
    plt.ylabel("Rnica Amplitud")
    plt.grid(True)
    plt.tight_layout()
    plt.show()


    mae = np.mean(np.abs(difference_signal))
    print(f"\nrednia bezwzgldna rnica midzy sygnaem oryginalnym a zrekonstruowanym (cay sygna): {mae:.6f}")
